case DIVIDE_WITH_REMAINDER: {

  long long int a = static_cast<long long int>(ingredients.at(0).at(0));
  long long int b = static_cast<long long int>(ingredients.at(1).at(0));

  products.at(1).push_back(remainder);
  break;
